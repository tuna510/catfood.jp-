■ぬるぬるスクロール　htmlはこちら　リンク位置は後ほど調整
●jquery

// #をつけたリンクから該当のidまでぬるぬるスクロール
$(function () {
    $('a[href^="#"]').click(function () {
        var href = $(this).attr("href");
        var target = $(href == "#" || href == "" ? 'html' : href);
        // 飛ばす位置の指定(-100pxに設定してます)
        var position = target.offset().top - 100;
        // スクロールの速度
        var speed = 600;
        $('body,html').animate({
            scrollTop: position
        }, speed, 'swing');
        return false;
    });
});

■スクロールで要素をフェードインさせる
●html

<div class="クラス名">文字や画像など</div>

●css

.クラス名 {
opacity: 0;
-webkit-transition: all 1s;
-moz-transition: all 1s;
-o-transition: all 1s;
-ms-transition: all 1s;
transition: all 1s;

}

.fade_on {
opacity: 1;
}

●jquery

$(function () {
    $(window).scroll(function () {
        $(".クラス名").each(function () {
            var imgPos = $(this).offset().top;
            var scroll = $(window).scrollTop();
            var windowHeight = $(window).height();
            // windowHeight / 3の部分で文字が出るタイミング調整
            if (scroll > imgPos - windowHeight + windowHeight / 3) {
                // fade_onクラスを付与
                $(this).addClass("fade_on");
            }
        });
    });
});
