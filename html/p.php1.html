■PHP

■配列
ひとつの変数に複数の値をまとめて保存できる。pythonのリストみたいな
$配列名 = array(値１, 値2, ・・・);のかたち
$names = array("Maria","Abbie","Bunyan");　// インデックス番号は0,1,2...
// 取り出し方
echo $names[0];　// 結果：Maria
echo $names[1];　// 結果：Abbie

// 値の追加、上書き
$names[] = "Oda";　// 配列の末尾に値を追加
$names[1] = "Yog-Sothoth";　// 値の上書き



■連想配列
インデックス番号では無くキーで管理　pythonの辞書みたいな
$配列名 = array("キー"=>"値1"、……);のかたち
$user = array(
    "name" => "ち",
    "age" => 23,
    "gender => "man"
);

// 取り出し方
echo $user["name"];　// 結果：ち

// 追加方法、計算
$user["attribute"] = "cute";　// キーと値の追加 =>ではない
$user["age"] ++;　// ageが24になる 例えば+=5なら+5



■foreach文
配列or連想配列に対して、先頭のデータから順に繰り返し処理を行うための命令文
// 配列
$names = array("Maria","Abbie","Bunyan");
foreach($names as $value) {　// foreach(配列か連想配列 as 値変数)
    echo $value;　// 繰り返したい処理
}　// 結果：Maria Abbie Bunyan

// 連想配列
$user = array(
    "name" => "ち",
    "age" => 23,
    "gender => "man"
);
foreach($user as $key as $value) {　// foreach(配列か連想配列 as キー変数 => 値変数)
    echo $key.':'.$value;　// 繰り返したい処理
}　// 結果：name:ち age:23 gender:man



■for文
for(①初期化 ; ②ループの条件 ; ④変数の更新) { ③繰り返し処理 }のかたち
// 1から100までを出力する
for($i=1; Si<=100; $1++) {
    echo $i.'<br>';　// .'<br>'で改行 下にecho'<br>';を書いても良い
}



■while文
// 1から100までを出力する
$i = 1;　// 初期化
while($i <= 100) {　// ループの条件
    echo $i.'<br>';　// 繰り返し処理
    $i++;　// 変数の更新
}



■break文
ループを強制的に中断する（一部だけ飛ばしたいときは下記continue文を使おう）
if文などの条件と合わせて組み合わせるのが一般的らしい
for($i=1; Si<=10; $1++) {
    if($i > 5) {
    break;　// この条件だと$iが6になった時点で終了する
}
    echo $i;　// 結果：12345
}



■continue文
ループの特定の週だけスキップする（丸ごと中断したい場合は上記break文を使おう）
for($i=1; Si<=10; $1++) {
    if($i % 3 == 0) {
    continue;　// この条件だと3の倍数だけスキップする
}
    echo $i;　// 結果：12457810
}
