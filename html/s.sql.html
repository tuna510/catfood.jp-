            ■テーブルの結合　がっちゃんこ

            内部結合

            

                
SELECT カラムA, カラムB,… FROM テーブル名A
    INNER JOIN テーブル名B ON 結合の条件;
    // 内部結合の場合、ベースとなるテーブルから、条件にマッチするレコードがないものは削除される

            

            外部結合

            

                
SELECT カラムA, カラムB,… FROM テーブル名A
    // LEFTは左側のテーブルを、RIGHTは右側のテーブルを軸にして結合
    LEFT(RIGHT) OUTER JOIN テーブル名B ON 結合の条件;

            ■ソート　ORDER BY

            

                
SELECT カラムA,カラムB,… FROM テーブル名 ORDER BY ソートしたいカラム 順番の指定;
// 順番の指定……ASC(昇順)、DESC(降順)

            

            例
            

                

            






            ■レコードの数指定　LIMIT

            

                
SELECT カラムA,カラムB,… FROM テーブル名 LIMIT 取り出したい数;
// ORDER BYでソートしてからやると良い感じに。LIMITつける位置は一番最後

            

            例
            

                

            






            ■取り出して名前をつける　AS

            

                
SELECT カラム AS つけたい名前 FROM テーブル名;

            

            例
            

                

            







            ■分岐　CASE

            

                
SELECT カラム CASE // 「CASE」と「END」で囲んだ中に条件を記載
    WHEN 条件 THEN 条件を満たしたら表示する内容
    // (他にも分岐させたい内容があればまたWHEN…で)
    ELSE 条件を満たしていなかったら表示する内容
    END
FROM テーブル名;

            

            例
            

                

            







            ■曖昧検索　LIKE

            

                
SELECT カラムA,カラムB,… FROM テーブル名 WHERE 検索に使うカラム LIKE 曖昧検索の条件;
// 例えば前に「山」がつく名前を探したいなら"山%"、後ろにつく名前なら"%山",
// 山がつくもの全てなら"%山%"と書いておく。曖昧部分が１文字だけなら%の代わりに_を使う。
// また、条件に一致しない物を探したいならLIKEではなくNOT LIKEを使う。
// それと検索条件に「％」とか「_」を入れたい場合は「\_」や「\%」にすればOK(エスケープ文字)

            

            大文字小文字の区別つけたい
            

                
SELECT カラムA,カラムB,… FROM テーブル名 WHERE 検索に使うカラム LIKE BINARY 曖昧検索の条件;
